    <script>
        document.addEventListener('DOMContentLoaded', function() {
            initializeApp();
            loadFormData('brainstorm', 'step2', 'brainstormStep2Input');
            
            // --- WORKING 'NEXT STEP' LOGIC ---
            document.getElementById('brainstormStep2Form').addEventListener('submit', function(e) {
                e.preventDefault();
                saveFormData('brainstorm', 'step2', 'brainstormStep2Input');
                window.location.href = 'brainstorm_step3.html';
            });

            // --- NEW 'PREVIOUS' BUTTON LOGIC ---
            const previousButton = document.getElementById('previousButton');
            if (previousButton) {
              previousButton.addEventListener('click', function(event) {
                event.preventDefault();
                console.log('PREVIOUS_BUTTON_CLICKED: Attempting to save data for brainstorm, step2...');
                saveFormData('brainstorm', 'step2', 'brainstormStep2Input');
                console.log('SAVE_COMPLETE: Navigating to brainstorm_step1.html...');
                window.location.href = 'brainstorm_step1.html';
              });
            }

            // --- Universal Home Button Logic ---
            const navHomeButton = document.getElementById('navHomeButton');
            if (navHomeButton) {
                navHomeButton.addEventListener('click', function() {
                    saveFormData('brainstorm', 'step2', 'brainstormStep2Input');
                    showHomeConfirmationModal();
                });
            }

            const modalGoHomeButton = document.getElementById('modalGoHomeButton');
            if(modalGoHomeButton) {
                modalGoHomeButton.addEventListener('click', function() {
                    goHome(); // goHome() is in main.js
                });
            }
        });
    </script>
